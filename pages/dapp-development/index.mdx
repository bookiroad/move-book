# 🏗️ dApp 개발 (dApp Development)

Movement Network에서 실제 운영 가능한 탈중앙화 애플리케이션을 구축하는 종합 가이드입니다. NFT부터 DeFi, 게임까지 다양한 dApp 카테고리별 심화 개발 과정을 다룹니다.

## 🎯 학습 목표

이 섹션을 완료하면:
- Movement Network의 주요 dApp 카테고리를 완전히 이해할 수 있습니다
- NFT 마켓플레이스와 컬렉션을 구축할 수 있습니다
- DeFi 프로토콜과 DEX를 개발할 수 있습니다
- 온체인 게임과 게임 이코노미를 설계할 수 있습니다

```move
// ✅ 보안 체크리스트
public fun secure_function(user: &signer, amount: u64) {
    // 1. 입력 검증
    assert!(amount > 0, E_INVALID_AMOUNT);
    assert!(amount <= MAX_AMOUNT, E_AMOUNT_TOO_LARGE);
    
    // 2. 권한 확인
    let user_addr = signer::address_of(user);
    assert!(is_authorized(user_addr), E_NOT_AUTHORIZED);
    
    // 3. 상태 검증
    assert!(is_system_active(), E_SYSTEM_PAUSED);
    
    // 4. 재진입 방지
    assert!(!is_locked(), E_REENTRANT_CALL);
    set_lock(true);
    
    // 5. 비즈니스 로직 실행
    execute_business_logic(user, amount);
    
    // 6. 락 해제
    set_lock(false);
}
```

### 가스 최적화 패턴
```move
// ✅ 가스 효율적인 패턴
public fun batch_operation(items: vector<Item>) {
    let results = vector::empty();
    let i = 0;
    let len = vector::length(&items);
    
    // 배치 처리로 가스 절약
    while (i < len) {
        let item = vector::borrow(&items, i);
        let result = process_item(item);
        vector::push_back(&mut results, result);
        i = i + 1;
    };
    
    // 한 번에 저장
    store_batch_results(results);
}

// ❌ 가스 비효율적인 패턴
public fun individual_operations(items: vector<Item>) {
    let i = 0;
    while (i < vector::length(&items)) {
        let item = vector::borrow(&items, i);
        let result = process_item(item);
        store_single_result(result); // 매번 저장 - 가스 낭비
        i = i + 1;
    }
}
```

```typescript
// dApp 성능 모니터링 시스템
class DAppMonitor {
  async trackTransaction(txHash: string) {
    const startTime = Date.now();
    
    try {
      const result = await client.waitForTransaction(txHash);
      const endTime = Date.now();
      
      this.recordMetric('transaction_success', {
        latency: endTime - startTime,
        gasUsed: result.gas_used,
        blockHeight: result.version
      });
    } catch (error) {
      this.recordMetric('transaction_error', {
        error: error.message,
        txHash
      });
    }
  }
  
  private recordMetric(event: string, data: any) {
    // 메트릭 저장 및 분석
    console.log(`[${event}]`, data);
  }
}
```
