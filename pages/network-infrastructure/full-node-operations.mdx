                        })

                    if peer_match:

                    tx_match = self.patterns['transaction'].search(line)
                    if tx_match:
                        stats['transaction_volume'].append({
                            'time': log_time,
                            'count': int(tx_match.group(1))
                        })
                
                except Exception as e:
